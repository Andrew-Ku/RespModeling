;--Блок процедур----------------
;--Процедура для поиска слудующего состояния ----------------
PROCEDURE ChooseNextStateProc(timeArg0,nextStateArg0,timeArg1,nextStateArg1,timeArg2,nextStateArg2,timeArg3,nextStateArg3,timeArg4,nextStateArg4,timeArg5,nextStateArg5,timeArg6,nextStateArg6,timeArg7,nextStateArg7)
BEGIN
TEMPORARY minTime,nextState;
minTime = timeArg0; nextState = nextStateArg0;
IF (minTime>timeArg1) THEN BEGIN minTime = timeArg1; nextState = nextStateArg1; END;
IF (minTime>timeArg2) THEN BEGIN minTime = timeArg2; nextState = nextStateArg2; END;
IF (minTime>timeArg3) THEN BEGIN minTime = timeArg3; nextState = nextStateArg3; END;
IF (minTime>timeArg4) THEN BEGIN minTime = timeArg4; nextState = nextStateArg4; END;
IF (minTime>timeArg5) THEN BEGIN minTime = timeArg5; nextState = nextStateArg5; END;
IF (minTime>timeArg6) THEN BEGIN minTime = timeArg6; nextState = nextStateArg6; END;
IF (minTime>timeArg7) THEN BEGIN minTime = timeArg7; nextState = nextStateArg7; END;
NextStateMat[1,1] = minTime; NextStateMat[1,2] = nextState;
Return (nextState);
END;

;--Процедура для коррекции времени ----------------
PROCEDURE CorrectStateTime(currentStateArg, endTimeArg)
BEGIN
TEMPORARY countState,result;
countState = 1;
sumStateTimes = 0;
WHILE (countState <= 9) DO BEGIN
sumStateTimes = sumStateTimes + StateTimeMat[1,countState];
countState = countState+1;
END;
StateTimeMat[1, currentStateArg + 1] = StateTimeMat[1, currentStateArg + 1] + endTimeArg-sumStateTimes;
 Return(StateTimeMat[1, currentStateArg + 1]);
END;

return (1);
